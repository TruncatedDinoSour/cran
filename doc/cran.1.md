# NAME

cran - an easy package manager for R written in bash

# SYNOPSIS

cran [OPTION] [ARGS]...

# Description

version, v, -v, ver

> Print version and exit

install, i, -i, ins

> Install packages

download, d, -d, dow

> Download package source

remove, r, -r, rm

> Remove/uninstall packages

list, l, -l, ls

> List installed packages

help, h, -h, hlp

> Print all commands

update, u, -u, upt

> Update all packages

updates, U, -U, upd

> Print all updatable packages

depends, D, -D, dep

> Print dependencies in cranfile
